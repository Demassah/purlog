#untuk combobox menampilkan dari 
tr_pros_detail : kondisi status_receive = 0
		 id_sro != 0
		 group_by = id_sro

SELECT `a`.`id_sro`, `a`.`date_create`, `a`.`id_user`, `b`.`id_courir`, `a`.`id_user`, `c`.`full_name`, `d`.`name_courir`
FROM (`tr_sro` a)
JOIN `tr_do` b ON `b`.`id_do` = `a`.`id_do`
JOIN `sys_user` c ON `c`.`user_id` = `a`.`id_user`
JOIN `ref_courir` d ON `d`.`id_courir` = `b`.`id_courir`
WHERE `a`.`id_sro` = '1'
AND `a`.`status` = '2'
GROUP BY `id_sro`
ORDER BY `a`.`id_sro` ASC
